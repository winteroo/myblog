function extend(source) {
  let target = null;
  if (typeof source === 'object') {
    target = Array.isArray(source) ? [] : {}
    for (let key in source) {
      if (source.hasOwnProperty(key)) {
        if (typeof source[key] !== 'object') {
          target[key] = source[key]

          target[key] = extend(source[key])

      }
    }
  } else {
    target = source
  }
  return target
}
